\begin{algorithm}
        \caption{Adding Binary Integers}
        \begin{algorithmic}
        \INPUT $A[1..n]$ and $B[1..n]$ representing two n-bit binary numbers $a$ and $b$.
        \OUTPUT $C[1..n+1]$ representing the result of $a + b$ .
        \STATE $carry = 0$
        \FOR {$i = n$ \DOWNTO $1$}
            \STATE $C[i + 1] = (A[i] + B[i] + carry) \bmod 2$
            \IF {$A[i] + B[i] + carry \ge 2$} 
                \STATE $carry = 1$
            \ELSE
                \STATE $carry = 0$
            \ENDIF
        \ENDFOR
        \STATE $C[1] = carry$
        \end{algorithmic}
        \end{algorithm}
