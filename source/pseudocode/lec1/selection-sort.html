\begin{algorithm}
        \caption{Selection-Sort($A[1..n]$)}
        \begin{algorithmic}
        \FOR {$i = 1$ \TO $n - 1$}
            \STATE $min = i$
            \FOR {$j = i + 1$ \TO $n$}
                \STATE \COMMENT{Find the index of $i^{th}$ smallest element}
                \IF {A[j] < A[min]}
                    \STATE $min = j$
                \ENDIF
            \ENDFOR
            \STATE \CALL{Swap}{$A[i], A[min]$}
        \ENDFOR
        \end{algorithmic}
        \end{algorithm}
