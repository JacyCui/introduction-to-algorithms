\begin{algorithm}
        \caption{Permute Without Identity}
        \begin{algorithmic}
        \INPUT an array $A$ of length $n$
        \OUTPUT a uniform random permutation of $A$
        \FOR{$i = 1$ \TO $n - 1$}
            \STATE swap $A[i]$ with $A[ $\CALL{Random}{$i + 1, n$} $]$
        \ENDFOR
        \end{algorithmic}
        \end{algorithm}
