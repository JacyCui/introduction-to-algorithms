\begin{algorithm}
        \caption{Randomize in Place - Modified for Proof}
        \begin{algorithmic}
        \INPUT An array $A$ of size $n$
        \OUTPUT A uniform randomized permutation of $A$
        \STATE swap $A[1]$ with $A[$ \CALL{Random}{$1, n$} $]$
        \FOR{$i = 2$ \TO $n$}
            \STATE swap $A[i]$ with $A[$ \CALL{Random}{$i, n$} $]$
        \ENDFOR
        \end{algorithmic}
        \end{algorithm}
