\begin{algorithm}
        \caption{Unbiased Random 0-1 Generator}
        \begin{algorithmic}
        \WHILE{\TRUE}
            \STATE $a =$ \CALL{BiasedRandom}{}
            \STATE $b =$ \CALL{BiasedRandom}{}
            \IF{$a \ne b$}
                \RETURN $a$
            \ENDIF
        \ENDWHILE
        \end{algorithmic}
        \end{algorithm}
