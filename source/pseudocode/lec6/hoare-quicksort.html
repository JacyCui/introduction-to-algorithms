\begin{algorithm}
        \caption{Quicksort Using Hoare Partition}
        \begin{algorithmic}
        \PROCEDURE{HoareQuicksort}{$A, p, r$}
            \IF{$p < r$}
                \STATE $q =$ \CALL{HoarePartition}{$A, p, r$}
                \STATE \CALL{HoareQuicksort}{$A, p, q$}
                \STATE \CALL{HoareQuicksort}{$A, q + 1, r$}
            \ENDIF
        \ENDPROCEDURE
        \end{algorithmic}
        \end{algorithm}
