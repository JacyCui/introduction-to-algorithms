\begin{algorithm}
        \caption{Quicksort Using Partition Considering Equal Elements}
        \begin{algorithmic}
        \PROCEDURE{Quicksort'}{$A, p, r$}
            \IF{$p < r$}
                \STATE $(q, t) =$ \CALL{Partition'}{$A, p, r$}
                \STATE \CALL{Quicksort'}{$A, p, q-1$}
                \STATE \CALL{Quicksort'}{$A, t+1, r$}
            \ENDIF
        \ENDPROCEDURE
        \end{algorithmic}
        \end{algorithm}
