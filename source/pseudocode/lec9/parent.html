\begin{algorithm}
        \caption{Get Parent of $x$ in Time Linear to $x$'s Number of Children}
        \begin{algorithmic}
        \PROCEDURE{Parent}{$x$}
            \STATE $p = x.left$-$child$
            \WHILE{\NOT $p.right$-$sibling$-$to$-$parent$}
                \STATE $p = p.right$-$sibling$
            \ENDWHILE
            \RETURN $p.right$-$sibling$
        \ENDPROCEDURE
        \end{algorithmic}
        \end{algorithm}
